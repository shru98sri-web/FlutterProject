Future delayedPrint(int second,String msg)
{
  final duration = Duration(seconds: second);
  return Future.delayed(duration).then((value)=>msg);
}
void main() async
{
  print("Haste");
  
  await delayedPrint(2,"makes").then ((status){
    print(status);
    
  });
 print("Waste"); 
}

Output
Haste 
makes
Waste

Future<String>delayFunction(){
 return Future.delayed(Duration(seconds:5),() =>"Weekday:");
 
}

void main() async {
  print("Weekend days are Saturday and Sunday");
  var weekend = ['Saturday','Sunday'];
  print(weekend);
  String data = await delayFunction();
  print(data);
  print("Weekdays are Monday to Friday");
  var weekday =['Monday','Tuesday','Wednesday','Thursday','Friday'];
  print(weekday);
}

Output

Weekend days are Saturday and Sunday
[Saturday,Sunday]
Weekday:
Weekdays are Monday to Friday
[Monday,Tuesday,Wednesday,Thursday,Friday]

  late double mark;

void main()			//function

3.Named Constructor

class Person {
  String? name;
  String? degree;
  int? age;

  Person(this.name, this.degree, this.age);
  // here Mobile() is a named constructor
  Person.namedConstructor(this.name, this.degree, [this.age = 0]);

  void displayPersonDetails() {
    print("Person name: $name.");
    print("Person degree: $degree.");
    print("Person price: $age");
  }
}

void main() {
  var person1 = Person("Samsung", "bsc", 20000);
  person1.displayPersonDetails();
  var person2 = Person.namedConstructor("Apple", "msc");
  person2.displayPersonDetails();
}

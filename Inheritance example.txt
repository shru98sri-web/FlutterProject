//Saving Account

package com.training;

public class SavingAccount extends BankAccount {
	
	private double currentBalance;
	
public SavingAccount(double currentBalance) {
	super();
	this.currentBalance = currentBalance;
}
	@Override
	public double deposit(double amt) {
		System.out.println("Saving account Deposit Called");
		return this.currentBalance +=amt;
		
	}
	
	@Override
	public double withdraw(double amt) {
		System.out.println("Saving account withdraw Called");
		return this.currentBalance +=amt;
		
	}
}

package com.training;

public class RecurringDeposit extends BankAccount {
	
	private double currentBalance;
	 public RecurringDeposit(double currentBalance) {
		 super();
		 this.currentBalance = currentBalance;
	 }

	@Override
	public double deposit(double amt) {
		return this.currentBalance+=amt;
	}
@Override 
public double withdraw(double amt) {
	if((this.currentBalance-amt)<5000) {
		System.out.println("Insufficient Balance");
		return this.currentBalance;
	}
	else {
		return this.currentBalance-=amt;
	}
}
}

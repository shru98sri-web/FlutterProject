class Product {
final String name;
final double price;
final int quantity;

  /// Creates a [Product] instance.
  ///
  /// [name] must not be null and represents the product's identifier.
  /// [price] must be non-negative and represents the cost per unit.
  /// [quantity] must be non-negative and defaults to 1 if not provided.
  /// Throws an [ArgumentError] if price or quantity are negative.

Product({
    required this.name,
    required this.price,
    this.quantity = 1,
  }) {
    if (price < 0) {
      throw ArgumentError('Product price cannot be negative.');
    }
    if (quantity < 0) {
      throw ArgumentError('Product quantity cannot be negative.');
    }
  }

  /// Calculates the total price for this specific product instance
  /// (price * quantity).
  double get itemTotalPrice => price * quantity;

  @override
  String toString() {
    return 'Product: $name (x${quantity}), Unit Price: \$${price.toStringAsFixed(2)}, Total: \$${itemTotalPrice.toStringAsFixed(2)}';
  }
}

/// Main entry point of the program.
void main() {
  List<Product> products = [];

  // Simulate user input or predefined products
  try {
    products = [
      Product(name: 'Laptop', price: 1200.00, quantity: 1),
      Product(name: 'Wireless Mouse', price: 25.99, quantity: 2),
      Product(name: 'Mechanical Keyboard', price: 79.99), // quantity defaults to 1
      Product(name: '4K Monitor', price: 350.00, quantity: 2),
      Product(name: 'USB-C Hub', price: 45.50),
    ];

    print('--- Product Inventory ---');
    print('Available products in the cart:\n');

    double grandTotal = 0.0;
    for (var product in products) {
      print(product);
      grandTotal += product.itemTotalPrice;
    }

    print('\n--- Order Summary ---');
    print('Total number of distinct product types: ${products.length}');
    print('Grand Total Price for all items: \$${grandTotal.toStringAsFixed(2)}');
    print('\nThank you for your order!');
  } on ArgumentError catch (e) {
    print('Error creating product: ${e.message}');
  } catch (e) {
    print('An unexpected error occurred: $e');
  }
}

18.import 'dart:math' as math;

class Circle {
  final double radius;

      : assert(radius >= 0, 'Radius cannot be negative.'),

    print(
        'Circle with radius ${radius.toStringAsFixed(2)} has an area of ${area.toStringAsFixed(2)}');
  }
}

    print('An unexpected error occurred: $e');
  }

20.class Student {
  final int age;
  final String rollNo;
  final double mark;

  // 1. Primary parameterized constructor
  
Student({required this.age, required this.rollNo, required this.mark});

  // 2. Named constructor: Initializes student with a default mark (e.g., 50.0)
  
  Student.withDefaultMark({required this.age, required this.rollNo})
      : mark = 50.0;

  // 3. Named constructor: Initializes student with only a roll number,
  // providing default values for age (e.g., 18) and mark (e.g., 60.0).
  
Student.withRollNoOnly({required this.rollNo})
      : age = 18,
        mark = 60.0;


  Student.defaultStudent()
      : age = 0,
        rollNo = 'UNKNOWN',
        mark = 0.0;		//Default Named Constructor

  // Method to display student details
  
	void show() {
    print('Age: $age, Roll No: $rollNo, Mark: $mark');
  }
}

void main() {
  // Using the primary parameterized constructor
  Student stud1 = Student(age: 27, rollNo: 'S001', mark: 80.5);
  print('--- Student 1 (Primary Constructor) ---');
  stud1.show(); // Output: Age: 27, Roll No: S001, Mark: 80.5

  // Using the named constructor 'withDefaultMark'
  Student stud2 = Student.withDefaultMark(age: 20, rollNo: 'S002');
  print('\n--- Student 2 (withDefaultMark Constructor) ---');
  stud2.show(); // Output: Age: 20, Roll No: S002, Mark: 50.0

  // Using the named constructor 'withRollNoOnly'
  Student stud3 = Student.withRollNoOnly(rollNo: '003');
  print('\n--- Student 3 (withRollNoOnly Constructor) ---');
  stud3.show(); // Output: Age: 18, Roll No: S003, Mark: 60.0

  // Using the named constructor 'defaultStudent'
  Student stud4 = Student.defaultStudent();
  print('\n--- Student 4 (defaultStudent Constructor) ---');
  stud4.show(); // Output: Age: 0, Roll No: UNKNOWN, Mark: 0.0
}

Why inherit?
In order to add functionality to a class we need to inherit. Class cannot be modified. Inheritance allows you to define a class that extends the functionality of another class.

Dart supports single inheritance. It means that a class can inherit from a single class. Dart doesn’t support multiple inheritance. In other words, a class cannot inherit from two or more classes.
